Create SQL query¶

We want Date, Donation_Amount, PFL, Metro_Type, Project_Cost, Status. This requires a join across three tables: donations, projects, schools. We will also filter donation amounts > 500 and state is Colorado and cost < 6000.

In [4]:
query = 'SELECT donations.Date, \
            donations.Donation_Amount, \
            schools.School_Percentage_Free_Lunch, \
            schools.School_Metro_Type, \
            projects.Project_Cost, \
            projects.Project_Current_Status \
        FROM donations, projects, schools \
        WHERE donations.Donation_Amount > 500 AND \
            donations.Project_ID = projects.Project_ID AND \
            projects.School_ID = schools.School_ID AND \
            schools.School_State = "Colorado" AND \
            projects.Project_Cost < 6000 \
        ORDER BY donations.Date'
dcdf = pd.read_sql_query(query, eng)

def prob_val(pfl, cost, metro_coef):
    val = reg6a.params['Intercept'] + reg6a.params['Cost']*3000 + reg6a.params['PFL']*pfl + metro_coef
    return expit(val)
