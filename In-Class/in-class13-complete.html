Data cleaning¶

Let's remove news headers. Up to the first occurrence of two blank lines. Also remove email addresses in signatures.

In [19]:
data = news_json.content.values.tolist()

data = [re.sub(r'\A(.+:\s+.+\n)+\n', '', entry) for entry in data]

data = [re.sub(r'\S+@\S+', '', entry) for entry in data]

In [28]:
import warnings

with warnings.catch_warnings():
    warnings.simplefilter("ignore")

    lda_model = gensim.models.ldamodel.LdaModel(corpus=data_id, id2word=id2word, num_topics=20,
                                           random_state=20191117, update_every=1, chunksize=100,
                                           passes=10, alpha='auto', per_word_topics=True)
